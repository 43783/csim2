
----------------------------------------------------------------------------------------------------------
	1. export collaborateurs (not used)
----------------------------------------------------------------------------------------------------------

SELECT 
	MV2_AFFECT.AFFECTSTARTDATE, MV2_PROJECT.IDPROJECTHESSO, MV2_AFFECT.IDCOLLABORATOR, MV2_COLLABORATOR.IDGESTAC, MV2_COLLABORATOR.COLLABORATORNAME, MV2_COLLABORATOR.COLLABORATORFIRSTNAME, MV2_COLLABORATOR.COLLABORATORCATEGORY, MV2_AFFECT.AFFECTRATE
FROM 
	(MV2_AFFECT INNER JOIN MV2_COLLABORATOR ON MV2_AFFECT.IDCOLLABORATOR = MV2_COLLABORATOR.IDCOLLABORATOR) INNER JOIN MV2_PROJECT ON MV2_AFFECT.IDPROJECT = MV2_PROJECT.IDPROJECT
WHERE 
	MV2_COLLABORATOR.IDSCHOOL=7;
	
----------------------------------------------------------------------------------------------------------
	2. export projects
----------------------------------------------------------------------------------------------------------

SELECT 
	MV2_PROJECT.IDPROJECTHESSO, MV2_PROJECT.PROJECTENVSTARTDATE, MV2_PROJECT.PROJECTENVENDDATE, MV2_PROJECT.PROJECTTITLE
FROM 
	MV2_PROJECT
WHERE 
	MV2_PROJECT.IDSCHOOL=7;

----------------------------------------------------------------------------------------------------------
	3. export AFG:
----------------------------------------------------------------------------------------------------------

SELECT 
	MV2_AGF.AGFDATE, MV2_PROJECT.IDPROJECTHESSO, MV2_AGF.AGF, MV2_AGF.AGFAMOUNT
FROM 
	MV2_AGF INNER JOIN MV2_PROJECT ON MV2_AGF.IDPROJECT = MV2_PROJECT.IDPROJECT
WHERE 
	MV2_PROJECT.IDSCHOOL=7;

----------------------------------------------------------------------------------------------------------
	4. export financement
----------------------------------------------------------------------------------------------------------

SELECT 
	MV2_FINANCIALPRESTATION.PRESTATIONSTARTDATE, MV2_PROJECT.IDPROJECTHESSO, MV2_PROJECTPARTNER.PARTNERNAME, MV2_FINANCIALPRESTATION.PROJECTFINANCIALAMOUNT
FROM 
	(MV2_PROJECT INNER JOIN MV2_FINANCIALPRESTATION ON MV2_PROJECT.IDPROJECT = MV2_FINANCIALPRESTATION.IDPROJECT) INNER JOIN MV2_PROJECTPARTNER ON MV2_FINANCIALPRESTATION.IDPROJECTPARTNERSOURCE = MV2_PROJECTPARTNER.IDPROJECTPARTNER
WHERE
	MV2_PROJECT.IDSCHOOL=7;
	
----------------------------------------------------------------------------------------------------------
	5. export outsourcing
----------------------------------------------------------------------------------------------------------

SELECT 
	MV2_OUTSOURCING.OUTSOURCINGSTARTDATE, MV2_PROJECT.IDPROJECTHESSO, MV2_PROJECTPARTNER.PARTNERNAME, MV2_OUTSOURCING.OUTSOURCINGREMARK, MV2_OUTSOURCING.OUTSOURCINGAMOUNT
FROM 
	(MV2_OUTSOURCING INNER JOIN MV2_PROJECT ON MV2_OUTSOURCING.IDPROJECT = MV2_PROJECT.IDPROJECT) INNER JOIN MV2_PROJECTPARTNER ON MV2_OUTSOURCING.IDPROJECTPARTNEROUTSOURCED = MV2_PROJECTPARTNER.IDPROJECTPARTNEROUTSOURCED
WHERE 
	MV2_PROJECT.IDSCHOOL=7;
